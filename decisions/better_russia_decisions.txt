country_decisions = {

	# ==========================================================
	# Pan-Slavic assimilation. Each decision converts the provinces
	# you OWN of the targeted culture(s) to their Slavic (_new)
	# counterpart (e.g. estonian -> estonian_new), keeping each
	# culture's own identity but placing it in the slavic group.
	# Only while your primary culture is in the slavic group.
	# Cost: 5 diplomatic power per converted province, paid per
	# province so you can never go below 0. The decision reappears
	# whenever you conquer a new matching non-slavic province.
	# ==========================================================

	br_assimilate_baltic = {
		potential = {
			culture_group = slavic
			any_owned_province = { culture_group = baltic }
		}
		allow = {
			dip_power = 5
		}
		effect = {
			custom_tooltip = br_assimilate_cost_tt
			every_owned_province = {
				limit = { culture = estonian }
				if = {
					limit = { owner = { dip_power = 5 } }
					change_culture = estonian_new
					owner = { add_dip_power = -5 }
				}
			}
			every_owned_province = {
				limit = { culture = lithuanian }
				if = {
					limit = { owner = { dip_power = 5 } }
					change_culture = lithuanian_new
					owner = { add_dip_power = -5 }
				}
			}
			every_owned_province = {
				limit = { culture = latvian }
				if = {
					limit = { owner = { dip_power = 5 } }
					change_culture = latvian_new
					owner = { add_dip_power = -5 }
				}
			}
		}
		ai_will_do = { factor = 0 }
	}
