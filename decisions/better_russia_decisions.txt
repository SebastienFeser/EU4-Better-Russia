country_decisions = {

	# ==========================================================
	# Pan-Slavic assimilation - one decision per culture group
	# (-200 DIP each) and per individual culture (-100 DIP each).
	# Only available while primary culture is in the slavic group
	# (russian_new etc.) AND we own at least one province of the
	# target culture. Converts only our OWN provinces of that
	# culture to russian_new. The decision reappears whenever we
	# conquer a new matching non-slavic province.
	# ==========================================================

	# ---------- Culture groups (-200 DIP) ----------

	br_assimilate_baltic = {
		potential = {
			culture_group = slavic
			any_owned_province = { culture_group = baltic }
		}
		allow = {
			dip_power = 5
		}
		effect = {
			custom_tooltip = br_assimilate_cost_tt
			every_owned_province = {
				limit = { culture_group = baltic }
				if = {
					limit = { owner = { dip_power = 5 } }
					change_culture = russian_new
					owner = { add_dip_power = -5 }
				}
			}
		}
		ai_will_do = { factor = 0 }
	}
